# 通过id查找 name

/**
 * 递归查 name
 * @param {Array} list 数组
 * @param {String} id  id
 */
getTneeName(list, id) {
  for (let i = 0; i < list.length; i++) {
    let a = list[i]
    if (a.id == id) {
      return a.name
    } else {
      if (a.children?.length) {
        let res = this.getTneeName(a.children, id)
        if (res) return res
      }
    }
  }
},

init(){
  let arr = [];//原数组
  let name =  this.getTneeName(this.arr, id);// 拿到的name
}

# 通过id查找children

/**
 * 递归查 children
 * @param {Array} arr 数组
 * @param {String} id  id
 */
getChilren(arr, id) {
  for (let index = 0; index < arr.length; index++) {
    let item = arr[index]
    if (item.id == id) {
      if (item?.children?.length) {
        return item?.children
      } else {
        return []
      }
    } else if (item?.children?.length > 0) {
      this.getChilren(item.children, id)
    }
  }
},

init(){
  let arr = [];//原数组
  let chilren =  this.getChilren(this.arr, id);// 拿到的children
}

# 递归修改参数

/**
 * 递归修改某个参数
 * @param {Array} list 数组
 */
setTree(list) {
  for (let i = 0; i < list.length; i++) {
    list[i].id = null;//这里是 id 置空
    // if(list[i].id=='xx'){
    //   list[i].disable = true;
    // }
    if (list[i].children) {
      this.setTree(list[i].children)
    }
  }
  return list
},

init(){
  let arr = [];//原数组
  let list =  this.setTree(this.arr);// 拿到的新tree
}
