# 命令相关

# 常用git命令

# 设置git用户和邮件

git config --global user.email "git 远程地址"

git config --global user.name "name"

# 提交代码

//提交代码流程
git add .

git commit -m "我是备注"

git push origin master

# 创建分支

//创建本地分支
git branch xxx

//删除本地分支
git branch -D xxx

//删除远程分支
git push origin --delete xxx

//切换本地分支
git checkout xxx

//创建一个分支，并切换
git checkout -b xxx

# 合并分支

TIP

开发分支（dev）上的代码达到上线的标准后，要合并到 master 分支

git checkout dev
git pull
git checkout master
git marge dev
git push -u origin master

# tag命令

// 打轻量标签
git tag v1.0.0

/**
 * 打附加标签
 * -a 指定附加标签，-m指定附加信息
*/
git tag -a <tag_name> -m "v0.1"

// 推送tag 到远程
git push origin V1.0.0

// 删除本地tag
git tag -d V1.0.0

// 删除远程tag
git push origin :refs/tags/V1.0.0

// 列出已有tag
git tag

// 切换tag
git checkout V1.0.0
